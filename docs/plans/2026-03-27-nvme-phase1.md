// SPDX-License-Identifier: GPL-2.0-or-later

//! Sans-I/O NVMe driver — Phase 1: controller init + Identify Controller.
//!
//! Uses the [`RegisterBank`] trait for all register access, enabling
//! full unit testing without hardware. The driver reads capabilities,
//! configures admin submission/completion queues, and issues admin
//! commands via 64-byte SQEs and 16-byte CQEs.

use super::register_bank::RegisterBank;

// ── NVMe Controller Registers (offsets from BAR0) ──────────────────

const REG_CAP_LO: usize = 0x00;    // Controller Capabilities (low 32)
const REG_CAP_HI: usize = 0x04;    // Controller Capabilities (high 32)
const REG_VS: usize = 0x08;        // Version
const REG_CC: usize = 0x14;        // Controller Configuration
const REG_CSTS: usize = 0x1C;      // Controller Status
const REG_AQA: usize = 0x24;       // Admin Queue Attributes
const REG_ASQ_LO: usize = 0x28;    // Admin Submission Queue Base (low 32)
const REG_ASQ_HI: usize = 0x2C;    // Admin Submission Queue Base (high 32)
const REG_ACQ_LO: usize = 0x30;    // Admin Completion Queue Base (low 32)
const REG_ACQ_HI: usize = 0x34;    // Admin Completion Queue Base (high 32)

// ── CC (Controller Configuration) bits ─────────────────────────────

/// CC value: EN=1, CSS=0(NVM), MPS=0(4K), IOSQES=6(64B), IOCQES=4(16B).
const CC_ENABLE: u32 = 0x0046_0001;
/// CC.EN bit mask.
const CC_EN: u32 = 1;

// ── CSTS (Controller Status) bits ──────────────────────────────────

/// CSTS.RDY bit mask.
const CSTS_RDY: u32 = 1;

// ── CAP field extraction ───────────────────────────────────────────

/// Maximum poll iterations for controller ready/disable.
const MAX_POLL_ITERATIONS: u32 = 100_000;

// ── Error type ─────────────────────────────────────────────────────

/// Errors from NVMe driver operations.
#[derive(Debug, Clone, PartialEq, Eq)]
pub enum NvmeError {
    /// Controller did not become ready or disabled within timeout.
    Timeout,
    /// NVM command set not supported (CAP.CSS bit 0 not set).
    UnsupportedCommandSet,
    /// Operation attempted in wrong state.
    InvalidState,
    /// Completion status indicates command failure.
    CommandFailed { status: u16 },
    /// No valid completion available (phase mismatch).
    NoCompletion,
}

// ── Driver state ───────────────────────────────────────────────────

#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub enum NvmeState {
    Uninitialized,
    Disabled,
    Enabled,
}

// ── Driver struct ──────────────────────────────────────────────────

pub struct NvmeDriver<R: RegisterBank> {
    bank: R,
    // Capabilities
    max_queue_entries: u16,
    doorbell_stride: u8,
    timeout_ms: u32,
    // Admin queue state
    admin_sq_tail: u16,
    admin_cq_head: u16,
    admin_cq_phase: bool,
    admin_sq_phys: u64,
    admin_cq_phys: u64,
    admin_queue_size: u16,
    // Command tracking
    next_cid: u16,
    // State
    state: NvmeState,
}

impl<R: RegisterBank> NvmeDriver<R> {
    // 64-bit register helpers
    fn read64(&self, offset_lo: usize) -> u64 {
        let lo = self.bank.read(offset_lo) as u64;
        let hi = self.bank.read(offset_lo + 4) as u64;
        lo | (hi << 32)
    }

    fn write64(&mut self, offset_lo: usize, value: u64) {
        self.bank.write(offset_lo, value as u32);
        self.bank.write(offset_lo + 4, (value >> 32) as u32);
    }

    /// Compute doorbell register offset for a queue.
    fn doorbell_offset(&self, qid: u16, is_cq: bool) -> usize {
        0x1000 + ((2 * qid as usize + is_cq as usize) * (4 << self.doorbell_stride))
    }

    /// Poll CSTS.RDY until it matches `expected`.
    fn poll_ready(&self, expected: bool) -> Result<(), NvmeError> {
        for _ in 0..MAX_POLL_ITERATIONS {
            let csts = self.bank.read(REG_CSTS);
            if ((csts & CSTS_RDY) != 0) == expected {
                return Ok(());
            }
        }
        Err(NvmeError::Timeout)
    }
}

    /// Initialize the NVMe controller: read capabilities, disable if running.
    ///
    /// Returns the driver in `Disabled` state ready for admin queue setup.
    pub fn init(mut bank: R) -> Result<Self, NvmeError> {
        // Read capabilities (64-bit)
        let cap_lo = bank.read(REG_CAP_LO) as u64;
        let cap_hi = bank.read(REG_CAP_HI) as u64;
        let cap = cap_lo | (cap_hi << 32);

        let mqes = (cap & 0xFFFF) as u16;
        let to = ((cap >> 24) & 0xFF) as u32;
        let dstrd = ((cap >> 32) & 0xF) as u8;
        let css = ((cap >> 37) & 0xFF) as u8;

        // Verify NVM command set supported
        if css & 1 == 0 {
            return Err(NvmeError::UnsupportedCommandSet);
        }

        let _vs = bank.read(REG_VS); // Read version (informational)

        let mut driver = Self {
            bank,
            max_queue_entries: mqes + 1,
            doorbell_stride: dstrd,
            timeout_ms: to * 500,
            admin_sq_tail: 0,
            admin_cq_head: 0,
            admin_cq_phase: true,
            admin_sq_phys: 0,
            admin_cq_phys: 0,
            admin_queue_size: 0,
            next_cid: 0,
            state: NvmeState::Uninitialized,
        };

        // Disable controller if running
        let cc = driver.bank.read(REG_CC);
        if cc & CC_EN != 0 {
            driver.bank.write(REG_CC, cc & !CC_EN);
        }
        driver.poll_ready(false)?;
        driver.state = NvmeState::Disabled;

        Ok(driver)
    }
```

    /// Configure admin submission/completion queues and enable the controller.
    ///
    /// The caller must provide physical addresses of zeroed DMA buffers:
    /// - `sq_phys`: `queue_size * 64` bytes for the submission queue
    /// - `cq_phys`: `queue_size * 16` bytes for the completion queue
    pub fn setup_admin_queue(
        &mut self,
        sq_phys: u64,
        cq_phys: u64,
        queue_size: u16,
    ) -> Result<(), NvmeError> {
        if self.state != NvmeState::Disabled {
            return Err(NvmeError::InvalidState);
        }

        let size = queue_size.min(self.max_queue_entries);

        // AQA: (CQ size - 1) << 16 | (SQ size - 1)
        let aqa = (((size - 1) as u32) << 16) | ((size - 1) as u32);
        self.bank.write(REG_AQA, aqa);

        // ASQ and ACQ (64-bit)
        self.write64(REG_ASQ_LO, sq_phys);
        self.write64(REG_ACQ_LO, cq_phys);

        // Enable: CC.EN=1, CSS=0(NVM), MPS=0(4K), IOSQES=6, IOCQES=4
        self.bank.write(REG_CC, CC_ENABLE);

        // Wait for CSTS.RDY
        self.poll_ready(true)?;

        self.admin_sq_phys = sq_phys;
        self.admin_cq_phys = cq_phys;
        self.admin_queue_size = size;
        self.admin_sq_tail = 0;
        self.admin_cq_head = 0;
        self.admin_cq_phase = true;
        self.state = NvmeState::Enabled;

        Ok(())
    }
```

```rust
    /// Build an Identify Controller admin command (opcode 0x06, CNS=0x01).
    ///
    /// The caller writes `sqe` to DMA at `admin_sq_phys + sq_offset`,
    /// then writes `doorbell_value` to the register at `doorbell_offset`.
    pub fn identify_controller(
        &mut self,
        data_phys: u64,
    ) -> Result<AdminCommand, NvmeError> {
        if self.state != NvmeState::Enabled {
            return Err(NvmeError::InvalidState);
        }

        let cid = self.next_cid;
        self.next_cid = self.next_cid.wrapping_add(1);

        // Build 64-byte SQE
        let mut sqe = [0u8; 64];
        // CDW0: opcode=0x06 (Identify), CID in bits 31:16
        let cdw0: u32 = 0x06 | ((cid as u32) << 16);
        sqe[0..4].copy_from_slice(&cdw0.to_le_bytes());
        // PRP1 (bytes 24-31): data buffer physical address
        sqe[24..32].copy_from_slice(&data_phys.to_le_bytes());
        // CDW10 (bytes 40-43): CNS=0x01 (identify controller)
        sqe[40..44].copy_from_slice(&1u32.to_le_bytes());

        let sq_offset = (self.admin_sq_tail as u64) * 64;

        // Advance tail with wrap
        self.admin_sq_tail = (self.admin_sq_tail + 1) % self.admin_queue_size;

        Ok(AdminCommand {
            sqe,
            sq_offset,
            doorbell_offset: self.doorbell_offset(0, false),
            doorbell_value: self.admin_sq_tail as u32,
        })
    }

    /// Check for a completion in the admin CQ.
    ///
    /// The caller reads 16 bytes from `admin_cq_phys + admin_cq_head * 16`
    /// and passes them in. Returns `None` if no valid completion (phase
    /// mismatch), or `Some(CompletionResult)` with the CQ doorbell to ring.
    pub fn check_completion(
        &mut self,
        cqe_bytes: &[u8; 16],
    ) -> Result<Option<CompletionResult>, NvmeError> {
        // Parse status field (bytes 14-15)
        let status_raw = u16::from_le_bytes(cqe_bytes[14..16].try_into().unwrap());
        let phase = (status_raw & 1) != 0;

        // Check phase bit
        if phase != self.admin_cq_phase {
            return Ok(None);
        }

        // Parse completion fields
        let result = u32::from_le_bytes(cqe_bytes[0..4].try_into().unwrap());
        let cid = u16::from_le_bytes(cqe_bytes[12..14].try_into().unwrap());
        let status = status_raw >> 1; // Bits 15:1 = status code

        // Advance head with wrap + phase inversion
        self.admin_cq_head = self.admin_cq_head + 1;
        if self.admin_cq_head >= self.admin_queue_size {
            self.admin_cq_head = 0;
            self.admin_cq_phase = !self.admin_cq_phase;
        }

        Ok(Some(CompletionResult {
            completion: Completion { cid, status, result },
            cq_doorbell_offset: self.doorbell_offset(0, true),
            cq_doorbell_value: self.admin_cq_head as u32,
        }))
    }
```

// ── Identify Controller response ───────────────────────────────────

/// Parsed fields from the 4 KiB Identify Controller data structure.
#[derive(Debug, Clone)]
pub struct IdentifyController {
    /// Serial Number (bytes 4-23, ASCII, space-padded).
    pub serial_number: [u8; 20],
    /// Model Number (bytes 24-63, ASCII, space-padded).
    pub model_number: [u8; 40],
    /// Firmware Revision (bytes 64-71, ASCII).
    pub firmware_rev: [u8; 8],
    /// Maximum Data Transfer Size (byte 77). In units of 2^(12 + MDTS) bytes.
    /// 0 = no limit (host page size).
    pub max_data_transfer: u8,
    /// Number of Namespaces (bytes 516-519).
    pub num_namespaces: u32,
}

/// Parse the 4 KiB Identify Controller response into structured fields.
pub fn parse_identify_controller(data: &[u8; 4096]) -> IdentifyController {
    let mut serial_number = [0u8; 20];
    serial_number.copy_from_slice(&data[4..24]);

    let mut model_number = [0u8; 40];
    model_number.copy_from_slice(&data[24..64]);

    let mut firmware_rev = [0u8; 8];
    firmware_rev.copy_from_slice(&data[64..72]);

    let max_data_transfer = data[77];

    let num_namespaces = u32::from_le_bytes(data[516..520].try_into().unwrap());

    IdentifyController {
        serial_number,
        model_number,
        firmware_rev,
        max_data_transfer,
        num_namespaces,
    }
}
